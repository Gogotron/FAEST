\begin{frame}{VOLE as commitments}
  In a VOLE, we may consider the parties hold:
  \begin{itemize}
    \item Sender: a polynomial $P(X)=uX+v$ \uncover<2->{\hfill $\leftarrow$ special case of degree 1}
    \item Receiver: an evaluation point $q:=P(\Delta)$
  \end{itemize}
  \pause
  \pause
  \vspace{1em}
  \begin{block}{Notation}
    $\left<u\right>_e^d$ represents:
    \begin{itemize}
      \item Sender: polynomial $P\in \mathbb{F}_{2^\lambda}[X]_{\scriptscriptstyle\leqslant d}$ with leading term $u\in \mathbb{F}_{2^e}$ 
      \item Receiver: $q=P(\Delta)\in \mathbb{F}_{2^\lambda}$
    \end{itemize}
  \end{block}
\end{frame}

\begin{frame}{Manipulating VOLE commitments}
  $$a,b\in \mathbb{F}_{2^e},\qquad C\in \mathbb{F}_{2^{e{\cdot}n}},\mathsf{cst}\in \mathbb{F}_{2^\lambda}$$
  \begin{center}
    \setlength{\tabcolsep}{10pt}
    \renewcommand{\arraystretch}{1.5}
    \begin{tabular}{|c|c|c|}
      \cline{2-3}
      \multicolumn{1}{c|}{} &
      Sender & Receiver \\
      \hline
      $\left<a+b\right>^d$ &
      \multicolumn{2}{|c|}{$\left<a\right>^d+\left<b\right>^d$} \\
      \hline
      $\left<a{\cdot}C\right>_{e{\cdot}n}^{d}$ &
      \multicolumn{2}{|c|}{$\left<a\right>_e^d{\cdot}C$} \\
      \hline
      $\left<\mathsf{cst}\right>^0$ &
      \multicolumn{2}{|c|}{$\mathsf{cst}$} \\
      \hline
      $\left<a\right>^{d+1}$ &
      $\left<a\right>^dX$ & $\left<a\right>^d\Delta$ \\
      \hline
    \end{tabular}
  \end{center}
  \begin{block}{Note}
    We can compute affine functions of commitments, and we can lift commitments from $\mathbb{F}_{2^e}^{n}$ to $\mathbb{F}_{2^{e{\cdot}n}}$.
  \end{block}
\end{frame}
