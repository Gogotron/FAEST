\begin{frame}{Combining small \$-VOLEs}
  \begin{alertblock}{Problem}
    GGM tree with $\lambda$ leafs yields VOLE with $\Delta\in \mathbb{F}_{2^\lambda}$ in $O(2^\lambda)$ operations.
  \end{alertblock}
  \pause
  \begin{block}{Solution}
    Combine $\tau$ small VOLEs with $\Delta\in \mathbb{F}_{2^n}$, in $O(\tau2^n)$ operations, for $\lambda=n\tau$.
  \end{block}
  \ \\
  \begin{columns}
    \begin{column}{0.7\textwidth}
      \begin{overprint}
        \onslide<3>
          $$q_1=u_1\Delta_1+v_1,\quad\ldots,\quad q_\tau=u_\tau\Delta_\tau+v_\tau$$
        \onslide<4>
          $$q'_1=u\Delta_1+v_1,\quad\ldots,\quad q'_\tau=u\Delta_\tau+v_\tau$$
        \onslide<5>
          $$q=u\Delta+v$$
      \end{overprint}
    \end{column}
    \hspace{-7em}
    \begin{column}{0.01\textwidth}
      \vspace{-2em}
      \begin{overprint}
        \onslide<3-4>
          $$\text{in }\mathbb{F}_{2^n}$$
        \onslide<5>
          $$\text{in }\mathbb{F}_{2^\lambda}$$
      \end{overprint}
    \end{column}
  \end{columns}
\end{frame}


\begin{frame}{Combining small \$-VOLEs}
  Multiple \$-VOLEs: $q_i=u_i\Delta_i+v_i\hfill i\in [1,\tau],\ \lambda=n\tau$\\
  $$q_i\in \mathbb{F}_{2^n}^{\hat{\ell}},\quad u_i\in \mathbb{F}_2^{\hat{\ell}},\quad\Delta_i\in \mathbb{F}_{2^n},\quad v_i\in \mathbb{F}_{2^n}^{\hat{\ell}}$$
  \ \\\ \\\pause
  Correction values: $c_i=u_0-u_i\hfill u=u_0$\\
  \hphantom{Correction values:} $q'_i=q_i+c_i\Delta_i$\\
  \ \\
  New corrected VOLEs: $q'_i=u\Delta_i+v_i$
\end{frame}


\begin{frame}{Combining small \$-VOLEs}
  We have: $q_i\in \mathbb{F}_{2^n}^{\hat{\ell}},\ u\in \mathbb{F}_2^{\hat{\ell}},\ \Delta_i\in \mathbb{F}_{2^n},\ v_i\in \mathbb{F}_{2^n}^{\hat{\ell}}$
  \hfill $i\in [1,\tau],\ \lambda=n\tau$\\
  $$q'_1=u\Delta_1+v_1,\quad\ldots,\quad q'_\tau=u\Delta_\tau+v_\tau$$
  \ \\\ \\\pause
  Concatenate bit representations:
  \vspace{-1em}
  $$q=[q'_1\ldots q'_\tau],\qquad\Delta=[\Delta_1\ldots\Delta_\tau],\qquad v=[v_1\ldots v_\tau]$$
  \ \\[1em]\pause
  \begin{center}
    \begin{tabular}{rl}
      $q'_i=u\Delta_i+v_i$ & verified in $\mathbb{F}_{2^n}$\\
      \pause
      $q'_i=u\Delta_i+v_i$ & verified bit by bit in $\mathbb{F}_2^n$\\
      \pause
      $q=u\Delta+v$ & verified bit by bit in $\mathbb{F}_2^{n\tau}=\mathbb{F}_2^\lambda$\\
      \pause
      $q=u\Delta+v$ & verified in $\mathbb{F}_{2^\lambda}$\\
    \end{tabular}
  \end{center}
\end{frame}


\begin{frame}{Consistency check}
  \begin{alertblock}{Warning}
    The receiver can't check that $c_i=u_0-u_i$.
    The sender could lie!\\
    We might not have $q=u\Delta+v$.
  \end{alertblock}
  \pause
  \ \\
  Random $\mathbb{F}_2$-\emph{linear} universal hash function $H:\mathbb{F}_2^{\hat{\ell}}\to\mathbb{F}_2^{\lambda+B}$\\
  Send: $\tilde{u}=Hu$, $\tilde{v}=Hv$ \\
  Check: $\tilde{q}=\tilde{u}\Delta+\tilde{v}$ with $\tilde{q}=Hq$\\
  \pause
  \ \\
  Reveals $(\lambda+B)$-bit linear function of $u\in \mathbb{F}_2^{\hat{\ell}}$.\\
  Construct longer VOLE: $\hat{\ell}=\ell+2\lambda\ \rightarrow\ \hat{\ell}=\ell+3\lambda+B$\\
  Discard extra $\lambda+B$ bits
\end{frame}
