\begin{frame}{Concatenating \$-VOLEs}
  GGM tree with $n$ leafs yields VOLE with $\Delta\in \mathbb{F}_{2^n}$ in $O(2^n)$ operations. \\
  \ \\\pause
  Multiple \$-VOLEs: $q_i=u_i\Delta_i+v_i\hfill i\in [0,\tau),\ \lambda=n\tau$\\
  \ \\\pause
  Correction values: $c_i=u_0-u_i$\\
  \hphantom{Correction values:} $q'_i\leftarrow q_i+c_i\Delta_i$\\
  New VOLEs: $q'_i=u\Delta_i+v_i\hfill u=u_0$\\
  \ \\\pause
  Concatenate bit representations:
  \vspace{-1em}
  $$v=[v_0\ldots v_{\tau-1}],\qquad\Delta=[\Delta_0\ldots\Delta_{\tau-1}],\qquad q=[q'_0\ldots q'_{\tau-1}]$$

  \begin{overprint}
    \onslide<4>
    $$q'_i=u\Delta_i+v_i\text{ verified in }\mathbb{F}_{2^n}$$
    \onslide<5>
    $$q'_i=u\Delta_i+v_i\text{ verified bit by bit in }\mathbb{F}_2^n$$
    \onslide<6>
    $$q=u\Delta+v\text{ verified bit by bit in }\mathbb{F}_2^{n\tau}=\mathbb{F}_2^\lambda$$
    \onslide<7>
    $$q=u\Delta+v\text{ verified in }\mathbb{F}_{2^\lambda}$$
  \end{overprint}
\end{frame}


\begin{frame}{Consistency check}
  \begin{alertblock}{Warning}
    The receiver can't check that $c_i=u_0-u_i$.
    The sender could lie!\\
    We might not have $q=u\Delta+v$.
  \end{alertblock}
  \pause
  \ \\
  Random $\mathbb{F}_2$-\emph{linear} universal hash function %
  \only<-2>{$H:\mathbb{F}_2^\ell\to\mathbb{F}_2^{\lambda+B}$}%
  \only<3->{$H:\mathbb{F}_2^{\hat{\ell}}\to\mathbb{F}_2^{\lambda+B}$}\\
  Send: $\tilde{u}=Hu$, $\tilde{v}=Hv$ \\
  Check: $\tilde{q}=\tilde{u}\Delta+\tilde{v}$ with $\tilde{q}=Hq$\\
  \pause
  \ \\
  Reveals $(\lambda+B)$-bit linear function of $u\in \mathbb{F}_2^\ell$.\\
  Use longer VOLE: $\hat{\ell}=\ell+\lambda+B$,\quad $u\in \mathbb{F}_2^{\hat{\ell}}$\\
  Discard extra $\lambda+B$ bits
\end{frame}
